1) Start from the app.js file generated at the end of demo1. Make sure you install nodemon, express and change "start":"nodemon app.js" in package.json file

2) Add  res.send('<h1>Hello from Express</h1>'); in the second app.use function. Reload the browser and you should see "Hello from Express!" displayed on it. If you check the network tab, you will see the content-type set to "text/html"
Express has done this for you.

const http = require('http');

const express = require('express'); // import express

const app = express(); // store express in a const by calling the express as a function

app.use((req, res, next) => {
    console.log('In the middleware');
    next(); // Allows the request to continue to the next middleware in line
});

app.use((req, res, next) => {
    console.log("In another middleware!");
    res.send('<h1>Hello from Express</h1>'); // added this
})
const server = http.createServer(app); // pass app as the paramete to the createServer function

server.listen(3010);
